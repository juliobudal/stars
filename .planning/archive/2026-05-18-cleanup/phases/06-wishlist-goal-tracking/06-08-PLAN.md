    ```ruby
    require "rails_helper"

    RSpec.describe "Kid Wishlist Flow", type: :system do
      let!(:family) { create(:family, name: "Família Teste") }
      let!(:parent_profile) { create(:profile, :parent, family: family, name: "Papai") }
      let!(:child) { create(:profile, :child, family: family, name: "Filhote", points: 50) }
      let!(:reward) { create(:reward, family: family, title: "LEGO Star Wars", cost: 100) }

      it "permite ao filho fixar uma meta e a vê no dashboard" do
        sign_in_as_child(child)
        visit kid_rewards_path

        within("##{ActionView::RecordIdentifier.dom_id(reward)}") do
          click_on "Definir como meta"
        end

        # Redirected to kid_rewards_path (or stays); navigate to dashboard
        visit kid_root_path

        expect(page).to have_content("Minha meta")
        expect(page).to have_content("LEGO Star Wars")
        expect(page).to have_content("50/100")
        expect(page).to have_content("Faltam")
        expect(page).to have_content("50") # remaining stars
      end

      it "permite ao filho remover a meta" do
        child.update!(wishlist_reward: reward)
        sign_in_as_child(child)
        visit kid_rewards_path

        within("##{ActionView::RecordIdentifier.dom_id(reward)}") do
          click_on "Remover meta"
        end

        visit kid_root_path
        expect(page).to have_content("Escolha um prêmio como meta")
        expect(page).not_to have_content("Minha meta")
      end

      it "limpa a meta automaticamente quando o filho resgata o prêmio fixado" do
        child.update!(points: 100, wishlist_reward: reward)
        sign_in_as_child(child)

        # Trigger the existing redeem flow at /kid/rewards
        # Mirror the existing redeem interaction exactly (spec/system/reward_redemption_flow_spec.rb).
        # The reward card opens the redeem modal; open_modal_and_click handles the modal lookup + click.
        visit kid_rewards_path
        open_modal_and_click("modal_#{ActionView::RecordIdentifier.dom_id(reward)}", "Sim, quero!")

        # After redeem, wishlist is cleared (Plan 06-05)
        expect(child.reload.wishlist_reward_id).to be_nil
      end

      it "mostra a meta do filho no dashboard do pai" do
        child.update!(wishlist_reward: reward)
        sign_in_as_parent(parent_profile)
        visit parent_root_path

        expect(page).to have_content("Meta:")
        expect(page).to have_content("LEGO Star Wars")
      end
    end
    ```

    Critical:
    - Title and `it` descriptions in Brazilian Portuguese (project convention — match `reward_redemption_flow_spec.rb` style "permite ao filho ...").
    - Use `ActionView::RecordIdentifier.dom_id(reward)` (NOT just `dom_id` — only ApplicationController/views have it as a top-level method). Required at the top level inside the spec.
    - Sign-in helpers: verify exact names by reading `spec/support/system_auth_helpers.rb` — analog uses `sign_in_as_child(child)` and `sign_in_as_parent(parent_profile)`.
    - For the auto-clear test: if the open_modal_and_click helper doesn't exist or the redeem flow needs different interaction, READ `spec/system/reward_redemption_flow_spec.rb` and replicate its EXACT redeem interaction. The key invariant is asserting `child.reload.wishlist_reward_id` is `nil` after redeem.
    - The "live update via Turbo Stream from parent task approval" scenario is intentionally OMITTED here — Capybara + Turbo Stream broadcasts in tests can be flaky and require Cable adapter coordination. The unit-level assertion is already covered in Plan 06-01 (Profile spec broadcast tests) and Plan 06-02 (service spec broadcast). System spec covers the full HTTP round-trip per scenario via fresh `visit` calls.
    - Avoid `sleep` — use Capybara's auto-wait via `have_content` matchers.
    - `let!` (eager) so records exist before the first sign-in/visit.
    - The variable `parent_profile` (not `parent`) avoids shadowing Ruby's `Kernel#parent` method in some contexts.
