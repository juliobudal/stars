  <action>
    Create `app/controllers/kid/wishlist_controller.rb`:

    ```ruby
    class Kid::WishlistController < ApplicationController
      include Authenticatable
      before_action :require_child!
      layout "kid"

      def create
        reward = Reward.where(family_id: current_profile.family_id).find(params[:reward_id])
        result = Profiles::SetWishlistService.call(profile: current_profile, reward: reward)

        if result.success?
          respond_to do |fmt|
            fmt.html { redirect_to kid_rewards_path, notice: "Meta atualizada!" }
            fmt.turbo_stream { head :ok } # broadcast already pushed via Profile callback
          end
        else
          redirect_to kid_rewards_path, alert: result.error
        end
      end

      def destroy
        result = Profiles::SetWishlistService.call(profile: current_profile, reward: nil)

        if result.success?
          respond_to do |fmt|
            fmt.html { redirect_to kid_rewards_path, notice: "Meta removida." }
            fmt.turbo_stream { head :ok }
          end
        else
          redirect_to kid_rewards_path, alert: result.error
        end
      end
    end
    ```

    Critical:
    - `Reward.where(family_id: current_profile.family_id).find(params[:reward_id])` — first IDOR layer. A foreign reward_id raises `RecordNotFound`, which `ApplicationController` already handles (verify by reading ApplicationController if unsure — falls back to Rails default 404 → 302 redirect handled by middleware in dev).
    - The service is the second IDOR layer (defense in depth) — but the controller scope means a malicious foreign reward_id never reaches the service.
    - Never write `current_profile.update(wishlist_reward_id: ...)` directly (CLAUDE.md C-1).
    - `head :ok` for turbo_stream because the Profile callback already pushed the replace; no need to render a stream response.
    - Brazilian Portuguese in flash: "Meta atualizada!" / "Meta removida." (CLAUDE.md C-4).
    - Class name uses Rails-style `Kid::WishlistController` (matches `Kid::RewardsController` convention — same single colon-pair namespace).
    - Pass `current_profile` (PIN-authenticated session); never accept a `:profile_id` param.

    Do NOT add a `before_action :set_reward` callback or any other helper — keep the file as terse as the analog `Kid::RewardsController`.

    ```ruby
    require "rails_helper"

    RSpec.describe "Kid::Wishlist", type: :request do
      let(:family) { create(:family) }
      let(:other_family) { create(:family) }
      let(:child) { create(:profile, :child, family: family, points: 30) }
      let(:reward) { create(:reward, family: family, cost: 100, title: "LEGO") }
      let(:foreign_reward) { create(:reward, family: other_family, cost: 100) }

      context "when signed in as the child" do
        before do
          host! "localhost"
          sign_in_as(child)
        end

        describe "POST /kid/wishlist" do
          it "sets wishlist_reward and redirects to kid_rewards_path" do
            expect {
              post kid_wishlist_path, params: { reward_id: reward.id }
            }.to change { child.reload.wishlist_reward_id }.from(nil).to(reward.id)

            expect(response).to redirect_to(kid_rewards_path)
            follow_redirect!
            expect(flash[:notice]).to match(/Meta atualizada/i)
          end

          it "rejects a cross-family reward_id (RecordNotFound short-circuits)" do
            expect {
              post kid_wishlist_path, params: { reward_id: foreign_reward.id }
            }.not_to change { child.reload.wishlist_reward_id }

            expect(response.status).to be_in([302, 404])
          end

          it "rejects a missing reward (RecordNotFound on find)" do
            expect {
              post kid_wishlist_path, params: { reward_id: 999_999 }
            }.not_to change { child.reload.wishlist_reward_id }
          end
        end

        describe "DELETE /kid/wishlist" do
          before { child.update!(wishlist_reward: reward) }

          it "clears wishlist_reward and redirects" do
            expect {
              delete kid_wishlist_path
            }.to change { child.reload.wishlist_reward_id }.from(reward.id).to(nil)

            expect(response).to redirect_to(kid_rewards_path)
            follow_redirect!
            expect(flash[:notice]).to match(/Meta removida/i)
          end
        end
      end

      context "when not signed in" do
        it "POST does not change wishlist and does not 200" do
          host! "localhost"
          expect {
            post kid_wishlist_path, params: { reward_id: reward.id }
          }.not_to change { child.reload.wishlist_reward_id }

          expect(response).not_to have_http_status(:ok)
        end
      end
    end
    ```

    Notes:
    - `sign_in_as(child)` is the kid-side helper from `spec/support/system_auth_helpers.rb:37` (verify exact name — analog `wallet_and_rewards_spec.rb:11` uses `sign_in_as(child)`). If the helper signature differs, mirror exactly what wallet_and_rewards_spec uses.
    - `host! "localhost"` is required — kid namespace expects localhost domain in tests.
    - `follow_redirect!` to access `flash[:notice]` after redirect.
    - The cross-family 404 test asserts on status `[302, 404]` because the rescue may render an error page or redirect — both are acceptable (key invariant: state did not change).
    - The unauth test asserts on `not have_http_status(:ok)` — covers both 302 (redirect) and 401/403 outcomes.
