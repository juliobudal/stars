    ```ruby
    require "rails_helper"

    RSpec.describe "PWA install integration", type: :request do
      describe "GET /manifest" do
        it "serves a valid expanded manifest" do
          get "/manifest"
          expect(response).to have_http_status(:ok)

          json = JSON.parse(response.body)
          expect(json["name"]).to eq("LittleStars")
          expect(json["lang"]).to eq("pt-BR")
          expect(json["start_url"]).to include("source=pwa")
          expect(json["icons"]).to be_an(Array)
          expect(json["icons"].map { |i| i["sizes"] }).to include("192x192", "512x512")
          expect(json["icons"].any? { |i| i["purpose"]&.include?("maskable") }).to be(true)
        end
      end

      describe "GET /service-worker" do
        it "serves the SW JS with the expected cache version" do
          get "/service-worker"
          expect(response).to have_http_status(:ok)
          expect(response.body).to include("littlestars-v1")
          expect(response.body).to include("addEventListener")
          expect(response.body).to include("offline.html")
        end
      end

      describe "kid layout" do
        let(:family) { create(:family) }
        let(:child)  { create(:profile, :child, family: family) }

        before do
          # Mirror the pattern from spec/requests/kid/wallet_and_rewards_spec.rb
          post "/family_session", params: { family_session: { email: family.email, password: "password" } }
          post "/profile_session", params: { profile_session: { profile_id: child.id, pin: child.raw_pin } }
        end

        it "renders both install components hidden" do
          get "/kid"
          expect(response).to have_http_status(:ok)
          expect(response.body).to match(/data-controller="ui--install-prompt--install-prompt".*hidden/m)
            .or match(/hidden.*data-controller="ui--install-prompt--install-prompt"/m)
          expect(response.body).to include("ui--ios-install-hint--ios-install-hint")
        end

        it "renders the manifest link tag in head" do
          get "/kid"
          expect(response.body).to include('rel="manifest"')
          expect(response.body).to include('name="theme-color"')
        end
      end

      describe "parent layout" do
        let(:family) { create(:family) }
        let(:parent) { create(:profile, :parent, family: family) }

        before do
          post "/family_session", params: { family_session: { email: family.email, password: "password" } }
          post "/profile_session", params: { profile_session: { profile_id: parent.id, pin: parent.raw_pin } }
        end

        it "renders both install components hidden" do
          get "/parent"
          expect(response).to have_http_status(:ok)
          expect(response.body).to include("ui--install-prompt--install-prompt")
          expect(response.body).to include("ui--ios-install-hint--ios-install-hint")
        end
      end
    end
    ```

    NOTES:
    - This is a `type: :request` spec, NOT `type: :system`. Why: the SW registration / install prompt UI flow involves browser-only events (`beforeinstallprompt`, `navigator.standalone`) that headless Chrome does not fire reliably. The request spec proves the SERVER-SIDE wiring (routes resolve + layouts render the components hidden). Browser-side behavior is verified manually + via Lighthouse in Task 2.
    - `child.raw_pin` and the family session helpers must match what existing specs do. If the project uses a different sign-in helper (`sign_in_as(child)`), use that instead. CONFIRM by reading `spec/requests/kid/wallet_and_rewards_spec.rb` before writing this spec.
    - If `family.email` doesn't exist, find the canonical sign-in approach.
